# Hyper Alpha Arena 前端架构

本文档详细解析前端架构、组件设计和开发规范。

## 技术栈

- **框架**: React 18 + TypeScript
- **构建工具**: Vite 4
- **UI 库**: shadcn/ui + Tailwind CSS
- **状态管理**: React Context API
- **图表库**: Chart.js, Lightweight Charts, Recharts
- **代码编辑**: Monaco Editor
- **国际化**: i18next
- **实时通信**: WebSocket

```typescript
// frontend/app/components/hyper-ai/HyperAiPage.tsx

export function HyperAiPage() {
  const [messages, setMessages] = useState<Message[]>([])
  const [isLoading, setIsLoading] = useState(false)
  
  const sendMessage = async (content: string) => {
    setIsLoading(true)
    
    // 添加用户消息
    const userMessage = { role: 'user', content }
    setMessages(prev => [...prev, userMessage])
    
    // 调用 API
    const response = await fetch('/api/hyper-ai/chat', {
      method: 'POST',
      body: JSON.stringify({ message: content })
    })
    
    // 流式读取响应
    const reader = response.body.getReader()
    const decoder = new TextDecoder()
    
    let aiMessage = { role: 'assistant', content: '' }
    
    while (true) {
      const { done, value } = await reader.read()
      if (done) break
      
      const chunk = decoder.decode(value)
      const events = chunk.split('\n\n')
      
      for (const event of events) {
        if (event.startsWith('data: ')) {
          const data = JSON.parse(event.slice(6))
          aiMessage.content += data.content
          setMessages(prev => [...prev.slice(0, -1), aiMessage])
        }
      }
    }
    
    setIsLoading(false)
  }
  
  return (
    <div className="flex flex-col h-screen">
      <ChatMessages messages={messages} />
      <ChatInput onSend={sendMessage} disabled={isLoading} />
    </div>
  )
}
```

```typescript
    }
```

```typescript
// frontend/app/lib/api.ts

const API_BASE = '/api'

export async function fetchAPI(endpoint: string, options?: RequestInit) {
  const token = Cookies.get('arena_token')
  
  const headers = {
    'Content-Type': 'application/json',
    ...(token && { 'Authorization': `Bearer ${token}` }),
    ...options?.headers
  }
  
  const response = await fetch(`${API_BASE}${endpoint}`, {
    ...options,
    headers
  })
  
  if (!response.ok) {
    const error = await response.json()
    throw new Error(error.detail || 'Request failed')
  }
  
  return response.json()
}
```

```typescript
// frontend/app/main.tsx

const ws = new WebSocket(`ws://${window.location.host}/ws`)

ws.onopen = () => {
  console.log('WebSocket connected')

  ws.send(JSON.stringify({
    type: 'bootstrap',
    username: 'default',
    initial_capital: 10000,
    trading_mode: 'paper'
  }))

ws.onmessage = (event) => {
  const msg = JSON.parse(event.data)
  
  switch (msg.type) {
    case 'bootstrap_ok':
      setUser(msg.user)
      setAccount(msg.account)
      break
    
    case 'snapshot':
      setOverview(msg.overview)
      setPositions(msg.positions)
      setOrders(msg.orders)
      setTrades(msg.trades)
      break
    
    case 'trade_update':
      setTrades(prev => [msg.trade, ...prev])
      toast.success('New trade executed!')
      break
    
    case 'asset_curve_update':
      setAllAssetCurves(msg.data)
      break
    
    case 'order_filled':
      toast.success('Order filled')
      // 刷新数据
      break
  }
}

// 发送请求
const placeOrder = (payload: any) => {
  ws.send(JSON.stringify({
    type: 'place_order',
    ...payload
  }))
}
```
